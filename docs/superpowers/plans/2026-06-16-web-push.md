# Web Push Notifications Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Browser push notifications via FCM web, reusing the existing push backend — a service worker for background system notifications, foreground in-app toasts, web token registration through the existing token API, and the `/me` notification-settings UI.

**Architecture:** A thin client layer on the dormant server slice. `firebaseConfig.ts` exposes a browser FCM `Messaging`; a service worker handles background notifications + click deep-links; `webPush.ts` registers the token via the existing `POST /api/me/push-token` (`platform: "web"`) and pipes foreground messages to a toast. No send-pipeline changes; one edit accepts `notifPrefs` in `/api/me` PATCH.

**Tech Stack:** Next.js 16, `firebase@11` (modular `firebase/messaging` + compat SW), Vitest.

**Spec:** `docs/superpowers/specs/2026-06-16-web-push-design.md`

```ts
/** Returns the FCM Messaging instance in supporting browsers, else null. Lazy + cached. */
export function getClientMessaging(): Promise<Messaging | null> {
  if (typeof window === "undefined") return Promise.resolve(null);
  if (!messagingPromise) {
```

```js
  messagingSenderId: "576145690934",
  appId: "1:576145690934:web:0650e4b648edc7332be397",
});

firebase.messaging(); // enables FCM auto-display of `notification` payloads in the background

// Open/focus the app at the notification's deep-link path on click.
self.addEventListener("notificationclick", (event) => {
  event.notification.close();
  const path = (event.notification.data && event.notification.data.path) || "/";
  event.waitUntil(
    clients.matchAll({ type: "window", includeUncontrolled: true }).then((wins) => {
      for (const w of wins) {
        if ("focus" in w) { w.navigate(path); return w.focus(); }
      }
      return clients.openWindow(path);
    })
  );
});
```

```ts
import { getClientMessaging } from "@/firebaseConfig";
import { getToken, onMessage, isSupported, type MessagePayload } from "firebase/messaging";

export async function isWebPushSupported(): Promise<boolean> {
  if (typeof window === "undefined") return false;
  if (!("Notification" in window) || !("serviceWorker" in navigator)) return false;
  return isSupported();
}

/** Requests permission, registers the SW, gets the FCM token, and stores it. Returns the permission. */
export async function registerWebPush(): Promise<NotificationPermission> {
  if (!(await isWebPushSupported())) return "denied";
  const permission = await Notification.requestPermission();
  if (permission !== "granted") return permission;

  const registration = await navigator.serviceWorker.register("/firebase-messaging-sw.js");
  const messaging = await getClientMessaging();
  if (!messaging) return permission;

  const token = await getToken(messaging, {
    vapidKey: process.env.NEXT_PUBLIC_FIREBASE_VAPID_KEY,
    serviceWorkerRegistration: registration,
  });
  if (token) {
    await fetch("/api/me/push-token", {
      method: "POST",
      headers: { "Content-Type": "application/json" },
      body: JSON.stringify({ token, platform: "web" }),
    });
  }
  return permission;
}

export async function unregisterWebPush(token: string): Promise<void> {
  await fetch("/api/me/push-token", {
    method: "DELETE",
    headers: { "Content-Type": "application/json" },
    body: JSON.stringify({ token }),
  }).catch(() => {});
}

/** Subscribe to foreground messages (app open). Returns an unsubscribe fn. */
export async function listenForeground(cb: (payload: MessagePayload) => void): Promise<() => void> {
  const messaging = await getClientMessaging();
  if (!messaging) return () => {};
  return onMessage(messaging, cb);
}
```

## Task 4: WebPushManager (silent re-register + foreground toast)

**Files:** Create `app/components/WebPushManager.tsx`; Modify `app/(client)/layout.tsx`

- [ ] **Step 1: Create the component**
```tsx
"use client";
import { useEffect, useState } from "react";
import { useSession } from "next-auth/react";
import { registerWebPush, listenForeground, isWebPushSupported } from "@/app/utils/webPush";

type Toast = { title: string; body: string; path: string } | null;

/** Browser-only: when permission is already granted, re-registers the token and shows
 * a foreground toast for incoming messages (background ones go through the service worker). */
export default function WebPushManager() {
  const { status } = useSession();
  const [toast, setToast] = useState<Toast>(null);

  useEffect(() => {
    if (status !== "authenticated") return;
    let unsub: (() => void) | undefined;
    let cancelled = false;
    (async () => {
      if (!(await isWebPushSupported())) return;
      if (typeof Notification === "undefined" || Notification.permission !== "granted") return;
      await registerWebPush().catch(() => {});
      unsub = await listenForeground((p) => {
        if (cancelled) return;
        setToast({
          title: p.notification?.title ?? "Notificación",
          body: p.notification?.body ?? "",
          path: (p.data?.path as string) ?? "/",
        });
      });
    })();
    return () => { cancelled = true; unsub?.(); };
  }, [status]);

  if (!toast) return null;
  return (
    <button
      onClick={() => { const path = toast.path; setToast(null); window.location.assign(path); }}
      className="fixed top-[max(1rem,env(safe-area-inset-top))] left-1/2 -translate-x-1/2 z-[80] w-[min(24rem,calc(100vw-2rem))] text-left rounded-2xl border border-[#00bfff]/30 bg-[#0a1929] shadow-2xl px-4 py-3 animate-overlay-in"
    >
      <p className="font-display text-sm font-bold text-[#00bfff]">{toast.title}</p>
      <p className="font-body text-sm text-gray-300">{toast.body}</p>
    </button>
  );
}
```
> `animate-overlay-in` already exists in `app/(client)/globals.css`. Auto-dismiss is intentionally omitted (YAGNI); tapping it navigates and clears. If you want auto-dismiss, that's a follow-up.

```tsx
"use client";
import { useEffect, useState } from "react";
import { registerWebPush, isWebPushSupported } from "@/app/utils/webPush";

type Prefs = { assignments: boolean; setlist: "all" | "assigned" | "off"; proposals: boolean; reminders: boolean };
const DEFAULTS: Prefs = { assignments: true, setlist: "all", proposals: true, reminders: true };

export default function NotificationSettings({ initial }: { initial?: Partial<Prefs> }) {
  const [prefs, setPrefs] = useState<Prefs>({ ...DEFAULTS, ...initial });
  const [perm, setPerm] = useState<NotificationPermission | "unsupported">("default");

  useEffect(() => {
    (async () => {
      if (!(await isWebPushSupported())) { setPerm("unsupported"); return; }
      setPerm(Notification.permission);
    })();
  }, []);

  async function save(next: Prefs) {
    setPrefs(next);
    await fetch("/api/me", { method: "PATCH", headers: { "Content-Type": "application/json" }, body: JSON.stringify({ notifPrefs: next }) });
  }
  async function enable() {
    const result = await registerWebPush();
    setPerm(result);
  }

  return (
    <section className="rounded-2xl border border-[#003572]/20 dark:border-[#00bfff]/15 p-5">
      <h3 className="font-display text-lg font-bold mb-1">Notificaciones</h3>
      {perm === "unsupported" ? (
        <p className="font-body text-sm text-gray-500 mb-4">Tu navegador no admite notificaciones push.</p>
      ) : perm === "granted" ? (
        <p className="font-body text-sm text-[#00bfff] mb-4">Notificaciones activadas en este dispositivo.</p>
      ) : perm === "denied" ? (
        <p className="font-body text-sm text-amber-400 mb-4">Notificaciones bloqueadas — habilítalas en los ajustes del navegador.</p>
      ) : (
        <button onClick={enable} className="font-label text-xs uppercase tracking-widest px-4 py-2 rounded-full border border-[#00bfff] text-[#00bfff] bg-[#00bfff]/10 mb-4">
          Activar notificaciones
        </button>
      )}
      <Toggle label="Asignaciones" on={prefs.assignments} set={(v) => save({ ...prefs, assignments: v })} />
      <SetlistChoice value={prefs.setlist} set={(v) => save({ ...prefs, setlist: v })} />
      <Toggle label="Propuestas" on={prefs.proposals} set={(v) => save({ ...prefs, proposals: v })} />
      <Toggle label="Recordatorios" on={prefs.reminders} set={(v) => save({ ...prefs, reminders: v })} />
    </section>
  );
}

function Toggle({ label, on, set }: { label: string; on: boolean; set: (v: boolean) => void }) {
  return (
    <div className="flex items-center justify-between py-2">
      <span className="font-body text-sm">{label}</span>
      <button onClick={() => set(!on)} aria-pressed={on} className={`px-3 py-1 rounded-full border text-xs font-label uppercase ${on ? "border-[#00bfff] text-[#00bfff] bg-[#00bfff]/10" : "border-[#003572]/25 text-gray-500"}`}>{on ? "Sí" : "No"}</button>
    </div>
  );
}
function SetlistChoice({ value, set }: { value: "all" | "assigned" | "off"; set: (v: "all" | "assigned" | "off") => void }) {
  const opts: { v: "all" | "assigned" | "off"; l: string }[] = [{ v: "all", l: "Todos" }, { v: "assigned", l: "Donde sirvo" }, { v: "off", l: "No" }];
  return (
    <div className="flex items-center justify-between py-2 gap-2">
      <span className="font-body text-sm shrink-0">Setlists</span>
      <div className="flex flex-wrap gap-1.5 justify-end">
        {opts.map((o) => (
          <button key={o.v} onClick={() => set(o.v)} aria-pressed={value === o.v} className={`px-3 py-1 rounded-full border text-xs font-label uppercase ${value === o.v ? "border-[#00bfff] text-[#00bfff] bg-[#00bfff]/10" : "border-[#003572]/25 text-gray-500"}`}>{o.l}</button>
        ))}
      </div>
    </div>
  );
}
```
